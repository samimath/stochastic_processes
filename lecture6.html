Simulation of branching process

branch_geom <- function(n,p) {
z <- c(1,rep(0,n))
for (i in 2:(n+1)) {
        z[i] <- sum(rgeom(z[i-1],p))
        }
        return(z) }

branch_custom <- function(n,alpha) {
    z <- c(1,rep(0,n))
    s <-length(alpha)-1
    for (i in 2:(n+1)) {
            z[i] <- sum( sample(0:s, z[i-1], replace=T, prob=alpha))
            }
            return(z) }

branch_poisson <- function(n,lam) {  
    z <- c(1,rep(0,n))
    for (i in 2:(n+1)) {
            z[i] <- sum(rpois(z[i-1],lam))
            }
            return(z)
}

Simulation of Normal random variable using MCMC (Metropolis-Hastings approach)

trials <- 500000
simlist <- numeric(trials)

normal_func <- function(x,mu,s){
  
  f <- 1/(s*sqrt(2*pi))*(exp(-((x-mu)/s)^2))
}
mu <-0
s<- 4

state <-mu

for (i in 2:trials){
  prop <- sample(c(state-1/s,state+1/s),size = 1,prob = c(0.5,0.5))
  # set numerator of acceptance function (evaluated at current state)
  pi_j <-normal_func(prop,mu,s)
  # set denominator of acceptance function (evaluated at proposed state)
  pi_i <-normal_func(state,mu,s)
  #acceptance function: 
  acc <- 0.5*(pi_j/pi_i)
  
  if (runif(1) < acc){
    
    state <- prop
    
  }
  else{
    state <-state
  }
  simlist[i] <- state
  
}
hist(simlist,xlab="",main="",prob=T)
curve(dnorm(x,mean = mu,sd = s),mu-10,mu+10,add=T,n = trials)
